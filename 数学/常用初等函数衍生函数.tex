\section*{第一组：指数函数与幂函数的乘除}

%--- 函数 f(x) = xe^x ---
\begin{tcolorbox}[
    enhanced,
    sidebyside, % 左右布局
    sidebyside align=center,
    colback=mygreen!5!white,
    colframe=mygreen!75!black,
    title=$f(x) = xe^x$,
    fonttitle=\bfseries,
]
    \begin{tikzpicture}
        \begin{axis}[
            width=\linewidth, height=6cm,
            axis lines=middle, axis line style={axiscolor, -{Stealth[length=2mm]}},
            xlabel={$x$}, ylabel={$y$},
            xmin=-3, xmax=2, ymin=-0.5, ymax=2,
            grid=major, grid style={gridcolor, dashed},
            samples=100,
        ]
        \addplot[mygreen, thick, domain=-3:2] {x*exp(x)};
        \node[circle, fill=mygreen, inner sep=1.5pt] at (axis cs:-1, {-1/exp(1)}) {};
        \node[above right, color=mygreen, font=\small] at (axis cs:-1, {-1/exp(1)}) {$(-1, -1/e)$};
        \end{axis}
    \end{tikzpicture}
    \tcblower % 切换到右侧内容
    \textbf{定义域：} $\mathbb{R}$ \\
    \textbf{导函数：} $f'(x) = (x+1)e^x$ \\
    \textbf{分析：} 令 $f'(x) = 0$, 得 $x=-1$。
    \begin{itemize}
        \item $x<-1$, $f'(x)<0$, 单调递减。
        \item $x>-1$, $f'(x)>0$, 单调递增。
    \end{itemize}
    \textbf{极小值点：} \textcolor{mygreen}{\textbf{$(-1, -1/e)$}}。
\end{tcolorbox}

        \node[circle, fill=myred, inner sep=1.5pt] at (axis cs:1, {1/exp(1)}) {};
        \node[above right, color=myred, font=\small] at (axis cs:1, {1/exp(1)}) {$(1, 1/e)$};
        \end{axis}
    \end{tikzpicture}
    \tcblower % 切换到右侧内容
    \textbf{定义域：} $\mathbb{R}$ \\
    \textbf{导函数：} $f'(x) = (1-x)e^{-x}$ \\
    \textbf{分析：} 令 $f'(x) = 0$, 得 $x=1$。
    \begin{itemize}
        \item $x<1$, $f'(x)>0$, 单调递增。
        \item $x>1$, $f'(x)<0$, 单调递减。
    \end{itemize}
    \textbf{极大值点：} \textcolor{myred}{\textbf{$(1, 1/e)$}}。
\end{tcolorbox}

    \textbf{分析：} 令 $f'(x) = 0$, 得 $x=1/e$。
    \begin{itemize}
        \item $0 < x < 1/e$, $f'(x)<0$, 单调递减。
        \item $x > 1/e$, $f'(x)>0$, 单调递增。
    \end{itemize}
    \textbf{极小值点：} \textcolor{mypurple}{\textbf{$(1/e, -1/e)$}}。
\end{tcolorbox}

\section*{第三组：经典模型函数}

%--- 函数 f(x) = e^x/x ---
\begin{tcolorbox}[
    enhanced,
    sidebyside, % 左右布局
    sidebyside align=center,
    colback=myblue!5!white,
    colframe=myblue!75!black,
    title=$f(x) = \frac{e^x}{x}$,
    fonttitle=\bfseries,
]
    \begin{tikzpicture}
        \begin{axis}[
            width=\linewidth, height=6cm,
            axis lines=middle, axis line style={axiscolor, -{Stealth[length=2mm]}},
            xlabel={$x$}, ylabel={$y$},
            xmin=-2, xmax=3, ymin=-2, ymax=8,
            grid=major, grid style={gridcolor, dashed},
            samples=100,
        ]
        \addplot[myblue, thick, domain=0.05:3] {exp(x)/x};
        \addplot[myblue, thick, domain=-2:-0.1] {exp(x)/x};
        \node[circle, fill=myblue, inner sep=1.5pt] at (axis cs:1, {exp(1)}) {};
        \node[below right, color=myblue, font=\small] at (axis cs:1, {exp(1)}) {$(1, e)$};
        \end{axis}
    \end{tikzpicture}
    \tcblower % 切换到右侧内容
    \textbf{定义域：} $\{x \in \mathbb{R} \mid x \neq 0\}$ \\
    \textbf{导函数：} $f'(x) = \frac{e^x(x-1)}{x^2}$ \\
    \textbf{分析：} 令 $f'(x) = 0$, 得 $x=1$。
    \begin{itemize}
        \item $x<1$ ($x \neq 0$), $f'(x)<0$, 单调递减。
        \item $x>1$, $f'(x)>0$, 单调递增。
    \end{itemize}
    \textbf{极小值点：} \textcolor{myblue}{\textbf{$(1, e)$}}。
\end{tcolorbox}

    \textbf{分析：} 令 $f'(x) = 0$, 得 $x=e$。
    \begin{itemize}
        \item $0<x<1$ 或 $1<x<e$, $f'(x)<0$, 单调递减。
        \item $x>e$, $f'(x)>0$, 单调递增。
    \end{itemize}
    \textbf{极小值点：} \textcolor{myorange}{\textbf{$(e, e)$}}。
\end{tcolorbox}
